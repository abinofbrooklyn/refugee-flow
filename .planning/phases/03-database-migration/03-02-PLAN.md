Create `scripts/seed.js` with the following structure. Each seeder function checks row count first (idempotent), transforms JSON data, and uses knex.batchInsert.

```javascript
require('dotenv').config();
const path = require('path');
const fs = require('fs');
const { round, uniqBy } = require('lodash');
const db = require('../server/database/connection');

const datasetsDir = path.join(__dirname, '../server/controllers/api/data/datasets');
const loadJSON = (file) => JSON.parse(fs.readFileSync(path.join(datasetsDir, file), 'utf8'));
const reduceGeo = (num) => round(num, 2);

async function seedWarEvents() {
  const count = (await db('war_events').count('id as n').first()).n;
  if (parseInt(count) > 0) { console.log('war_events: already seeded'); return; }

  const warData = loadJSON('war_all.json');
  const rows = [];
  for (const yearEntry of warData) {
    for (const [quarter, events] of Object.entries(yearEntry.value)) {
      // Sort by fat DESC, then dedup on rounded lat,lng (same logic as warReducer in dataProcessors.js)
      const sorted = [...events].sort((a, b) => b.fat - a.fat);
      const deduped = uniqBy(sorted, i => `${reduceGeo(i.lat)},${reduceGeo(i.lng)}`);
      for (const e of deduped) {
        rows.push({
          id: e.id,
          year: parseInt(yearEntry.Year),
          quarter,
          fat: e.fat,
          int: e.int,
          evt: e.evt,
          cot: JSON.stringify(e.cot),
          lat: reduceGeo(e.lat),
          lng: reduceGeo(e.lng),
        });
      }
    }
  }
  await db.batchInsert('war_events', rows, 1000);
  console.log(`war_events: inserted ${rows.length} rows`);
}

async function seedAsyApplications() {
  const count = (await db('asy_applications').count('id as n').first()).n;
  if (parseInt(count) > 0) { console.log('asy_applications: already seeded'); return; }

  const asyData = loadJSON('asy_application_all.json');
  const rows = [];
  // asyData is an object with year keys: { "2010": { q1: [...], q2: [...], q3: [...], q4: [...] }, ... }
  for (const [yearStr, quarters] of Object.entries(asyData)) {
    for (const [quarter, records] of Object.entries(quarters)) {
      for (const rec of records) {
        rows.push({
          year: parseInt(yearStr),
          quarter,
          origin: rec.Origin,
          destination: rec.destination,
          value: typeof rec.Value === 'number' ? rec.Value : null,
          record_id: rec.id,
        });
      }
    }
  }
  await db.batchInsert('asy_applications', rows, 1000);
  console.log(`asy_applications: inserted ${rows.length} rows`);
}

async function seedRouteDeaths() {
  const count = (await db('route_deaths').count('id as n').first()).n;
  if (parseInt(count) > 0) { console.log('route_deaths: already seeded'); return; }

  const deaths = loadJSON('route_death.json');
  const rows = deaths.map(d => ({
    id: parseInt(d.id),
    date: d.date,
    quarter: d.quarter,
    year: parseInt(d.year),
    dead: d.dead,
    missing: d.missing,
    dead_and_missing: d.dead_and_missing,
    cause_of_death_display: d.cause_of_death_displayText,
    cause_of_death: d.cause_of_death,
    location: d.location,
    description: d.description,
    source: d.source,
    // Precision reduce lat/lng but store as string to preserve API contract
    lat: String(reduceGeo(parseFloat(d.lat))),
    lng: String(reduceGeo(parseFloat(d.lng))),
    route: d.route,
    route_display: d.route_displayText,
    source_url: d.source_url,
  }));
  await db.batchInsert('route_deaths', rows, 1000);
  console.log(`route_deaths: inserted ${rows.length} rows`);
}

async function seedIbcCrossings() {
  const count = (await db('ibc_crossings').count('id as n').first()).n;
  if (parseInt(count) > 0) { console.log('ibc_crossings: already seeded'); return; }

  const ibcData = loadJSON('IBC_all.json');
  const rows = [];
  // ibcData is object keyed by route name. Each route has array of nationality records.
  // Each record has year keys ("2009"-"2018") with {q1:N,q2:N,q3:N,q4:N} plus Route, BorderLocation, NationalityLong
  for (const [routeName, nationalities] of Object.entries(ibcData)) {
    for (const natRecord of nationalities) {
      const borderLocation = natRecord.BorderLocation;
      const nationality = natRecord.NationalityLong;
      // Iterate year keys (skip non-year keys: Route, BorderLocation, NationalityLong)
      for (const [key, quarterObj] of Object.entries(natRecord)) {
        if (['Route', 'BorderLocation', 'NationalityLong'].includes(key)) continue;
        const year = parseInt(key);
        if (isNaN(year)) continue;
        for (const [quarter, value] of Object.entries(quarterObj)) {
          rows.push({
            route: routeName,
            nationality,
            border_location: borderLocation,
            year,
            quarter,
            value: value != null ? value : null,
          });
        }
      }
    }
  }
  await db.batchInsert('ibc_crossings', rows, 1000);
  console.log(`ibc_crossings: inserted ${rows.length} rows`);
}

async function seedCountryRoutes() {
  const count = (await db('country_routes').count('id as n').first()).n;
  if (parseInt(count) > 0) { console.log('country_routes: already seeded'); return; }

  const countries = loadJSON('country_route_list.json');
  const rows = countries.map(c => ({
    country: c.country,
    routes: JSON.stringify(c.route),
  }));
  await db.batchInsert('country_routes', rows, 1000);
  console.log(`country_routes: inserted ${rows.length} rows`);
}

async function main() {
  try {
    console.log('Starting seed...');
    await seedWarEvents();
    await seedAsyApplications();
    await seedRouteDeaths();
    await seedIbcCrossings();
    await seedCountryRoutes();
    // war_notes: table exists but seeded empty (ACLED data comes in Phase 4)
    console.log('Seed complete.');
  } catch (err) {
    console.error('Seed failed:', err);
    process.exit(1);
  } finally {
    await db.destroy();
  }
}

main();
```

Key transformation details:
- war_events: lat/lng precision-reduced via `round(num, 2)`, deduped per year+quarter via `uniqBy(sorted, i => roundedLat,roundedLng)`, sorted by fat DESC before dedup (keeps highest fatality entry)
- route_deaths: lat/lng parsed to float, rounded to 2 decimals, stored back as string (e.g., "41.244376" becomes "41.24") to preserve the API string contract
- asy_applications: Value field may be mixed type in source — store as integer if numeric, null otherwise
- ibc_crossings: flattened from nested {route: [{year: {quarter: value}}]} to flat rows (route, nationality, year, quarter, value)
- country_routes: route array stored as JSONB
- war_notes: table left empty (Phase 4 ACLED ingestion will populate)
