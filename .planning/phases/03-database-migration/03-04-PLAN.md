     for (let i = 0; i < retries; i++) {
       try {
         await db.raw('SELECT 1');
         return;
       } catch (err) {
         if (i === retries - 1) throw err;
         await new Promise(r => setTimeout(r, 1000));

4. Create `server/__tests__/geo.test.js`:
   ```javascript
   const { db, waitForDb, closeDb } = require('./helpers/testDb');

   beforeAll(async () => { await waitForDb(); }, 15000);
   afterAll(async () => { await closeDb(); });

   describe('Geo precision (DB-03)', () => {
     it('war_events lat values have at most 2 decimal places', async () => {
       // Cast to text and check for 3+ digits after decimal point
       const bad = await db.raw(
         "SELECT lat FROM war_events WHERE lat::text ~ '\\\\.[0-9]{3,}' LIMIT 1"
       );
       expect(bad.rows.length).toBe(0);
     });

     it('war_events lng values have at most 2 decimal places', async () => {
       const bad = await db.raw(
         "SELECT lng FROM war_events WHERE lng::text ~ '\\\\.[0-9]{3,}' LIMIT 1"
       );
       expect(bad.rows.length).toBe(0);
     });

     it('route_deaths lat values are precision-reduced strings', async () => {
       // lat is TEXT; check no value has more than 2 decimal digits
       const bad = await db.raw(
         "SELECT lat FROM route_deaths WHERE lat ~ '\\\\.[0-9]{3,}' LIMIT 1"
       );
       expect(bad.rows.length).toBe(0);
     });

     it('route_deaths lng values are precision-reduced strings', async () => {
       const bad = await db.raw(
         "SELECT lng FROM route_deaths WHERE lng ~ '\\\\.[0-9]{3,}' LIMIT 1"
       );
       expect(bad.rows.length).toBe(0);
     });
   });

   describe('War events dedup', () => {
     it('no duplicate (lat,lng) within same year+quarter', async () => {
       const dupes = await db.raw(`
         SELECT year, quarter, lat, lng, COUNT(*) as cnt
         FROM war_events
         GROUP BY year, quarter, lat, lng
         HAVING COUNT(*) > 1
         LIMIT 5
       `);
       expect(dupes.rows.length).toBe(0);
     });
   });

   describe('MongoDB removal (DB-01)', () => {
     it('no mongoose in package.json', () => {
       const pkg = require('../../package.json');
       expect(pkg.dependencies).not.toHaveProperty('mongoose');
       expect(pkg.devDependencies || {}).not.toHaveProperty('mongoose');
     });

     it('server starts without MONGODB_URI in env', async () => {
       // If we got this far with waitForDb(), server module loaded without mongoose
       // Just verify the connection module exports a knex instance
       const connection = require('../database/connection');
       expect(typeof connection.raw).toBe('function'); // Knex instance has .raw()
     });
   });
   ```
