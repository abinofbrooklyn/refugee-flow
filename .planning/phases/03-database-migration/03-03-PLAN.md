// GET /data/reduced_war_data — returns [{Year, value:{q1:[],q2:[],q3:[],q4:[]}}]
// App-layer reconstruction from flat rows (simpler than SQL aggregation)
const findReducedWar = async () => {
  const rows = await db('war_events')
    .select('id', 'year', 'quarter', 'fat', 'int', 'evt', 'cot', 'lat', 'lng')
    .orderBy([
      { column: 'year', order: 'asc' },
      { column: 'quarter', order: 'asc' },
      { column: 'fat', order: 'desc' },
    ]);

  const byYear = {};
  for (const row of rows) {
    const yearStr = String(row.year);
    if (!byYear[yearStr]) {
      byYear[yearStr] = { Year: yearStr, value: { q1: [], q2: [], q3: [], q4: [] } };
    }
    byYear[yearStr].value[row.quarter].push({
      id: row.id,
      fat: row.fat,
      int: row.int,
      evt: row.evt,
      cot: row.cot,  // JSONB auto-parsed by pg driver
      lat: parseFloat(row.lat),  // decimal comes back as string from pg, convert to number
      lng: parseFloat(row.lng),
    });
  }
  return Object.values(byYear);
};

// GET /data/asy_application_all — returns [{"2010":{q1:[],q2:[],q3:[],q4:[]},...}]
// CRITICAL: response is wrapped in a single-element array
const findAsyApplicationAll = async () => {
  const rows = await db('asy_applications')
    .select('year', 'quarter', 'origin', 'destination', 'value', 'record_id')
    .orderBy(['year', 'quarter']);

  const result = {};
  for (const row of rows) {
    const yearStr = String(row.year);
    if (!result[yearStr]) result[yearStr] = { q1: [], q2: [], q3: [], q4: [] };
    result[yearStr][row.quarter].push({
      Origin: row.origin,
      Value: row.value,
      id: row.record_id,
      destination: row.destination,
    });
  }
  return [result];  // WRAPPED IN ARRAY — matches current behavior
};

// GET /data/route_death — returns flat array, all values as strings
const findRouteDeath = async () => {
  const rows = await db('route_deaths')
    .select(
      'id', 'date', 'quarter', 'year',
      'dead', 'missing', 'dead_and_missing',
      'cause_of_death_display', 'cause_of_death',
      'location', 'description', 'source',
      'lat', 'lng', 'route', 'route_display', 'source_url'
    );

  // Map column names back to original JSON field names and cast all to strings
  return rows.map(r => ({
    id: String(r.id),
    date: r.date,
    quarter: r.quarter,
    year: String(r.year),
    dead: r.dead,
    missing: r.missing,
    dead_and_missing: r.dead_and_missing,
    cause_of_death_displayText: r.cause_of_death_display,  // column name differs from JSON key
    cause_of_death: r.cause_of_death,
    location: r.location,
    description: r.description,
    source: r.source,
    lat: r.lat,     // already TEXT in DB
    lng: r.lng,     // already TEXT in DB
    route: r.route,
    route_displayText: r.route_display,  // column name differs from JSON key
    source_url: r.source_url,
  }));
};

// GET /data/route_IBC — returns object keyed by route name
// Each route value is array of nationality records with year keys containing quarter objects
const findRouteIbc = async () => {
  const rows = await db('ibc_crossings')
    .select('route', 'nationality', 'border_location', 'year', 'quarter', 'value')
    .orderBy(['route', 'nationality', 'year', 'quarter']);

  // Rebuild the nested structure: { routeName: [{ "2009":{q1:N,...}, Route, BorderLocation, NationalityLong }] }
  const routeMap = {};
  for (const row of rows) {
    if (!routeMap[row.route]) routeMap[row.route] = {};
    const natKey = row.nationality;
    if (!routeMap[row.route][natKey]) {
      routeMap[row.route][natKey] = {
        Route: row.route,
        BorderLocation: row.border_location,
        NationalityLong: row.nationality,
      };
    }
    const yearStr = String(row.year);
    if (!routeMap[row.route][natKey][yearStr]) {
      routeMap[row.route][natKey][yearStr] = {};
    }
    routeMap[row.route][natKey][yearStr][row.quarter] = row.value;
  }

  // Convert nationality map to arrays per route
  const result = {};
  for (const [routeName, nationalities] of Object.entries(routeMap)) {
    result[routeName] = Object.values(nationalities);
  }
  return result;
};

// GET /data/route_IBC_country_list — returns [{country, route:[]}]
const findRouteIbcCountryList = async () => {
  const rows = await db('country_routes').select('country', 'routes');
  return rows.map(r => ({
    country: r.country,
    route: r.routes,  // JSONB auto-parsed to array by pg driver
  }));
};

module.exports = {
  findWarNote,
  findReducedWar,
  findAsyApplicationAll,
  findRouteDeath,
  findRouteIbcCountryList,
  findRouteIbc,
};
