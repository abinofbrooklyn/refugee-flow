1. Create `server/controllers/admin/adminController.js`:

   **csvPreview(req, res):**
   - Uses `parse` from `csv-parse/sync` with `{ columns: true, skip_empty_lines: true, trim: true }`
   - Parses `req.file.buffer` (multer memory storage)
   - Returns `res.json({ rows, count: rows.length })`
   - On parse error: `res.status(400).json({ error: err.message })`

   **csvCommit(req, res):**
   - Accepts `req.body` with `{ rows, target }` where target is one of: 'war_events', 'asy_applications', 'route_deaths'
   - Validates target is in allowed list: `['war_events', 'asy_applications', 'route_deaths']`
   - If target not in list: `res.status(400).json({ error: 'Invalid target table. Allowed: war_events, asy_applications, route_deaths' })`
   - For each row, apply `reduceGeoPercision` to lat/lng fields if present (parseFloat first, then reduce to 2 dp)
   - Uses `db(target).insert(rows).onConflict().ignore()` in batches of 500
   - Logs via `logIngestion({ source: 'csv', status: 'success', rowsAffected: rows.length, startedAt })`
   - Returns `res.json({ inserted: rows.length })`
   - On error: logs error and returns `res.status(500).json({ error: err.message })`

   **triggerIngestion(req, res):**
   - Reads `req.params.source` — must be one of 'acled', 'unhcr', 'iom'
   - Maps to runner function: `{ acled: runAcledIngestion, unhcr: runUnhcrIngestion, iom: runIomIngestion }`
   - If unknown source: `res.status(400).json({ error: 'Unknown source. Allowed: acled, unhcr, iom' })`
   - Runs the ingestion function (await)
   - Returns `res.json({ status: 'complete', source })`
   - On error: `res.status(500).json({ error: err.message, source })`

   Export: `module.exports = { csvPreview, csvCommit, triggerIngestion };`

2. Create `server/routes/adminRoute.js`:
   ```javascript
   const express = require('express');
   const multer = require('multer');
   const adminAuth = require('../middleware/adminAuth');
   const { csvPreview, csvCommit, triggerIngestion } = require('../controllers/admin/adminController');

   const upload = multer({ storage: multer.memoryStorage() });
   const router = express.Router();

   router.use(express.json());
   router.use(adminAuth);

   router.post('/csv/preview', upload.single('file'), csvPreview);
   router.post('/csv/commit', csvCommit);
   router.post('/trigger/:source', triggerIngestion);

   module.exports = router;
   ```

   ```javascript
   if (require.main === module) {
     const cron = require('node-cron');
     const { runAcledIngestion } = require('./ingestion/acledIngestion');
     const { runUnhcrIngestion } = require('./ingestion/unhcrIngestion');
     const { runIomIngestion } = require('./ingestion/iomIngestion');

     // Staggered weekly schedules at 02:00 server time
     cron.schedule('0 2 * * 1', () => runAcledIngestion().catch(err => console.error('[ACLED cron]', err.message)));   // Monday
     cron.schedule('0 2 * * 3', () => runUnhcrIngestion().catch(err => console.error('[UNHCR cron]', err.message)));   // Wednesday
     cron.schedule('0 2 * * 5', () => runIomIngestion().catch(err => console.error('[IOM cron]', err.message)));       // Friday

     app.listen(process.env.PORT);
   }
   ```
