One-time backfill script to normalize all existing route_deaths rows in the database. Uses the same normalizeRow and deduplicateRows functions from iomNormalizer.js.

```javascript
// scripts/normalizeRouteDeaths.js
// One-time script to normalize existing route_deaths data in-place.
// Run: node scripts/normalizeRouteDeaths.js
//
// This must run BEFORE simplifying dataController.js.
// After running, the database contains clean normalized route values.

require('dotenv').config();
const db = require('../server/database/connection');
const { normalizeRow, deduplicateRows } = require('../server/ingestion/iomNormalizer');

async function run() {
  console.log('Loading all route_deaths rows...');
  const rows = await db('route_deaths').select('*');
  console.log(`Found ${rows.length} rows`);

  // Normalize all rows
  const normalized = rows.map(r => {
    const norm = normalizeRow(r);
    // Strip internal flags
    const { _wasFallback, _rawRoute, ...clean } = norm;
    return { original: r, normalized: clean };
  });

  // Update rows where route, lat, or lng changed
  let updated = 0;
  for (const { original, normalized } of normalized) {
    if (normalized.route !== original.route ||
        normalized.lat !== original.lat ||
        normalized.lng !== original.lng ||
        normalized.route_display_text !== original.route_display_text) {
      await db('route_deaths').where({ id: original.id }).update({
        route: normalized.route,
        route_display_text: normalized.route,
        lat: normalized.lat,
        lng: normalized.lng,
      });
      updated++;
    }
  }
  console.log(`Updated ${updated} of ${rows.length} rows`);

  // Remove duplicates: find rows that deduplicateRows would filter out
  const allNormalized = normalized.map(n => n.normalized);
  const deduped = deduplicateRows(allNormalized);
  const dedupedIds = new Set(deduped.map(r => r.id));
  const duplicateIds = allNormalized.filter(r => !dedupedIds.has(r.id)).map(r => r.id);

  if (duplicateIds.length > 0) {
    console.log(`Removing ${duplicateIds.length} duplicate rows...`);
    await db('route_deaths').whereIn('id', duplicateIds).del();
    console.log(`Deleted ${duplicateIds.length} duplicates`);
  } else {
    console.log('No duplicates found');
  }

  const finalCount = await db('route_deaths').count('* as count').first();
  console.log(`Final row count: ${finalCount.count}`);
  process.exit(0);
}

run().catch(err => {
  console.error('Normalization failed:', err);
  process.exit(1);
});
```
